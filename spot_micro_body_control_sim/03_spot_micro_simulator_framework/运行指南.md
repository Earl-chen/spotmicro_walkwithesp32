创建一个运行脚本：

```bash
cat > run_all.sh << 'EOF'
#!/bin/bash
echo "======================================================================"
echo "SpotMicro Walk步态算法 - 一键测试"
echo "======================================================================"

echo ""
echo "【1/4】运行综合测试..."
python3 test_walk_gait.py

echo ""
echo "【2/4】验证占空比..."
python3 verify_duty_cycle.py

echo ""
echo "【3/4】绘制轨迹图..."
python3 plot_gait_trajectory_v2.py

echo ""
echo "【4/4】生成步态动画..."
python3 gait_animation_demo_v6.py

echo ""
echo "======================================================================"
echo "✅ 所有任务完成！"
echo "======================================================================"
echo "生成的文件："
ls -lh *.png *.gif 2>/dev/null
EOF

chmod +x run_all.sh
./run_all.sh
```
