```python
import websockets
import json

async def test_websocket_latency():
    uri = "ws://192.168.1.7:12580/ws"
```

### 3. 并发连接测试

**测试脚本** (`test_concurrent_connections.py`):
```python
import asyncio
import aiohttp
import time

async def make_request(session, url, semaphore):
    async with semaphore:
        try:
            start_time = time.time()
            async with session.get(url) as response:
                await response.text()
                end_time = time.time()
                return end_time - start_time, response.status
        except Exception as e:
            return None, str(e)

async def test_concurrent_connections():
    base_url = "http://192.168.1.7:12580"
    concurrent_requests = [10, 20, 50, 100]
    
    for num_requests in concurrent_requests:
        print(f"\n测试 {num_requests} 个并发连接:")
        
        semaphore = asyncio.Semaphore(num_requests)
        async with aiohttp.ClientSession() as session:
            tasks = []
            for i in range(num_requests):
                url = f"{base_url}/api/pos/x?value={i % 201 - 100}"
                task = make_request(session, url, semaphore)
                tasks.append(task)
            
            start_time = time.time()
            results = await asyncio.gather(*tasks)
            end_time = time.time()
            
            successful = [r for r in results if r[1] == 200]
            failed = [r for r in results if r[1] != 200]
            
            print(f"总耗时: {(end_time - start_time):.2f}s")
            print(f"成功请求: {len(successful)}")
            print(f"失败请求: {len(failed)}")
            if successful:
                avg_time = sum(r[0] for r in successful) / len(successful)
                print(f"平均响应时间: {avg_time * 1000:.2f}ms")

if __name__ == "__main__":
    asyncio.run(test_concurrent_connections())
```

## 🔍 实时控制延迟优化

### 1. 延迟测量

**测量工具** (`latency_monitor.py`):
```python
import asyncio
import websockets
import json
import time
import numpy as np
import matplotlib.pyplot as plt

class LatencyMonitor:
    def __init__(self, uri):
        self.uri = uri
        self.latencies = []
        self.timestamps = []
    
    async def monitor_latency(self, duration=60):
        """监控指定时间内的延迟"""
        async with websockets.connect(self.uri) as websocket:
            start_time = time.time()
            
            while time.time() - start_time < duration:
                # 发送姿态控制指令
                send_time = time.time()
                message = {
                    "type": "pose6",
                    "data": {
                        "x": 0.05 * np.sin(send_time),
                        "y": 0.0,
                        "z": -0.10,
                        "roll": 0.0,
                        "pitch": 0.0,
                        "yaw": 0.0
                    },
                    "timestamp": int(send_time * 1000)
                }
                
                await websocket.send(json.dumps(message))
                response = await websocket.recv()
                receive_time = time.time()
                
                latency = (receive_time - send_time) * 1000
                self.latencies.append(latency)
                self.timestamps.append(send_time)
                
                await asyncio.sleep(0.05)  # 20Hz控制频率
    
    def analyze_latency(self):
        """分析延迟数据"""
        latencies = np.array(self.latencies)
        
        print(f"延迟统计分析:")
        print(f"样本数量: {len(latencies)}")
        print(f"平均延迟: {np.mean(latencies):.2f}ms")
        print(f"中位数延迟: {np.median(latencies):.2f}ms")
        print(f"最小延迟: {np.min(latencies):.2f}ms")
        print(f"最大延迟: {np.max(latencies):.2f}ms")
        print(f"标准差: {np.std(latencies):.2f}ms")
        print(f"95%分位数: {np.percentile(latencies, 95):.2f}ms")
        print(f"99%分位数: {np.percentile(latencies, 99):.2f}ms")
        
        # 延迟分布统计
        low_latency = np.sum(latencies < 50)
        medium_latency = np.sum((latencies >= 50) & (latencies < 100))
        high_latency = np.sum(latencies >= 100)
        
        print(f"\n延迟分布:")
        print(f"低延迟 (<50ms): {low_latency} ({low_latency/len(latencies)*100:.1f}%)")
        print(f"中延迟 (50-100ms): {medium_latency} ({medium_latency/len(latencies)*100:.1f}%)")
        print(f"高延迟 (>100ms): {high_latency} ({high_latency/len(latencies)*100:.1f}%)")
    
    def plot_latency(self):
        """绘制延迟图表"""
        plt.figure(figsize=(12, 8))
        
        # 延迟时间序列
        plt.subplot(2, 2, 1)
        plt.plot(self.timestamps, self.latencies)
        plt.title('延迟时间序列')
        plt.xlabel('时间 (s)')
        plt.ylabel('延迟 (ms)')
        plt.grid(True)
        
        # 延迟直方图
        plt.subplot(2, 2, 2)
        plt.hist(self.latencies, bins=50, alpha=0.7)
        plt.title('延迟分布直方图')
        plt.xlabel('延迟 (ms)')
        plt.ylabel('频次')
        plt.grid(True)
        
        # 累积分布函数
        plt.subplot(2, 2, 3)
        sorted_latencies = np.sort(self.latencies)
        p = np.arange(1, len(sorted_latencies) + 1) / len(sorted_latencies)
        plt.plot(sorted_latencies, p * 100)
        plt.title('延迟累积分布函数')
        plt.xlabel('延迟 (ms)')
        plt.ylabel('累积概率 (%)')
        plt.grid(True)
        
        # 延迟箱线图
        plt.subplot(2, 2, 4)
        plt.boxplot(self.latencies)
        plt.title('延迟箱线图')
        plt.ylabel('延迟 (ms)')
        plt.grid(True)
        
        plt.tight_layout()
        plt.savefig('latency_analysis.png', dpi=300)
        plt.show()

async def main():
    monitor = LatencyMonitor("ws://192.168.1.7:12580/ws")
    
    print("开始延迟监控，持续60秒...")
    await monitor.monitor_latency(60)
    
    print("分析延迟数据...")
    monitor.analyze_latency()
    
    print("生成延迟图表...")
    monitor.plot_latency()

if __name__ == "__main__":
    asyncio.run(main())
```

### 3. 性能基准测试

**基准测试脚本** (`benchmark_test.py`):
```python
import asyncio
import websockets
import json
import time
import statistics

async def benchmark_test():
    """性能基准测试"""
    uri = "ws://192.168.1.7:12580/ws"
    
    # 测试场景
    scenarios = [
        {"name": "低频控制", "frequency": 1, "duration": 30},
        {"name": "中频控制", "frequency": 10, "duration": 30},
        {"name": "高频控制", "frequency": 20, "duration": 30},
        {"name": "极高频控制", "frequency": 50, "duration": 10},
    ]
    
    results = {}
    
    for scenario in scenarios:
        print(f"\n执行 {scenario['name']} 测试...")
        
        async with websockets.connect(uri) as websocket:
            latencies = []
            start_time = time.time()
            
            while time.time() - start_time < scenario['duration']:
                # 发送控制指令
                send_time = time.time()
                message = {
                    "type": "pose6",
                    "data": {
                        "x": 0.05,
                        "y": 0.0,
                        "z": -0.10,
                        "roll": 0.0,
                        "pitch": 0.0,
                        "yaw": 0.0
                    },
                    "timestamp": int(send_time * 1000)
                }
                
                await websocket.send(json.dumps(message))
                response = await websocket.recv()
                receive_time = time.time()
                
                latency = (receive_time - send_time) * 1000
                latencies.append(latency)
                
                # 控制频率
                await asyncio.sleep(1.0 / scenario['frequency'])
            
            # 统计结果
            results[scenario['name']] = {
                'count': len(latencies),
                'avg_latency': statistics.mean(latencies),
                'max_latency': max(latencies),
                'min_latency': min(latencies),
                'p95_latency': statistics.quantiles(latencies, n=20)[18],  # 95%分位数
                'success_rate': 100.0  # 假设所有请求都成功
            }
    
    # 输出结果
    print("\n=== 性能基准测试结果 ===")
    print(f"{'测试场景':<12} {'请求数':<8} {'平均延迟':<10} {'最大延迟':<10} {'95%延迟':<10} {'成功率':<8}")
    print("-" * 70)
    
    for name, result in results.items():
        print(f"{name:<12} {result['count']:<8} {result['avg_latency']:<10.2f} "
              f"{result['max_latency']:<10.2f} {result['p95_latency']:<10.2f} "
              f"{result['success_rate']:<8.1f}%")

if __name__ == "__main__":
    asyncio.run(benchmark_test())
```
